cmake_minimum_required(VERSION 3.2)

project(console-window)

set(CMAKE_CXX_STANDARD 17)

add_executable(window_test)

target_sources(window_test
  PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src/main.cc
  PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src/window.cc
  PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src/line.cc
  PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src/triangle.cc
)

target_include_directories(window_test PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/include)
